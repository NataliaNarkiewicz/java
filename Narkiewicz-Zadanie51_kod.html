<script>

let liczba_nar = prompt("Podaj liczbę n =", "");
let n_nar = parseInt(liczba_nar);

let silnia_nar = 1;

if (n_nar === 0) {
    silnia_nar = 1;
}
else if (n_nar > 0) {
    for (let i_nar = 1; i_nar <= n_nar; i_nar++) {
        silnia_nar = silnia_nar * i_nar;
    }
}
else {
    document.write("Silnia nie jest zdefiniowana dla liczb ujemnych!");
}

if (n_nar >= 0) {
    document.write("n = " + n_nar + "<br>");
    document.write(n_nar + "! = " + silnia_nar + "<br>");
}

</script>
