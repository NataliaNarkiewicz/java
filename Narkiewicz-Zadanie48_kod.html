<script>

do {
    var liczba1_nar = prompt("wysokość =", "");
    var h_nar = parseFloat(liczba1_nar);
}
while (h_nar <= 0);

do {
    var liczba1_nar = prompt("podstawę =", "");
    var a_nar = parseFloat(liczba1_nar);
}
while (!(a_nar > 0));

var Pole_nar = a_nar * h_nar / 2;

document.write("Pole = " + Pole_nar.toFixed(2) + " cm²<br>");

</script>
