Kod programu:

<script>
let ile_razy_nar = 12;       
let wielkosc_liter_nar = 41;  

let dana_nar = prompt("Podaj liczbę:");

document.write("ilość wykonania pętli = " + ile_razy_nar + "<br>");
document.write("wielkość liter = " + wielkosc_liter_nar + "px<br><br>");

if (dana_nar % 2 === 0) {
    for (let i_nar = 1; i_nar <= ile_razy_nar; i_nar++) {
        document.write("<span style='color:blue; font-size:" + wielkosc_liter_nar + "px;'>Narkiewicz</span><br>");
    }
} else {
    for (let i_nar = 1; i_nar <= ile_razy_nar; i_nar++) {
        document.write("<span style='color:green; font-size:" + wielkosc_liter_nar + "px;'>Natalia</span><br>");
    }
}
</script>
