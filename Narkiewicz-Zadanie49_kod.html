<script>

do {

    var liczba1_nar = prompt("Podaj bok a =", "");
    var a_nar = parseFloat(liczba1_nar);

    liczba1_nar = prompt("Podaj bok b =", "");
    var b_nar = parseFloat(liczba1_nar);

    liczba1_nar = prompt("Podaj bok c =", "");
    var c_nar = parseFloat(liczba1_nar);

}
while (
    !(a_nar > 0 && b_nar > 0 && c_nar > 0 &&
      a_nar + b_nar > c_nar &&
      a_nar + c_nar > b_nar &&
      b_nar + c_nar > a_nar)
);

document.write("Podane boki tworzą trójkąt.<br>");
document.write("a = " + a_nar + "<br>");
document.write("b = " + b_nar + "<br>");
document.write("c = " + c_nar + "<br>");

</script>
